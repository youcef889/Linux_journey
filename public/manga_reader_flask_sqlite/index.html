2. app/init.py
# 

from flask import Flask
from flask_sqlalchemy import SQLAlchemy
from flask_login import LoginManager
from flask_migrate import Migrate
from config import Config

db = SQLAlchemy()
migrate = Migrate()
login_manager = LoginManager()
login_manager.login_view = 'auth.login'

def create_app(config_class=Config):
    app = Flask(__name__)
    app.config.from_object(config_class)

    db.init_app(app)
    migrate.init_app(app, db)
    login_manager.init_app(app)

    from app.auth import bp as auth_bp
    app.register_blueprint(auth_bp, url_prefix='/auth')

    from app.main import bp as main_bp
    app.register_blueprint(main_bp)

    # Create upload directories if they don't exist
    import os
    for folder in [app.config['UPLOAD_FOLDER'], 
                  app.config['MANGA_COVERS'], 
                  app.config['MANGA_PAGES']]:
        if not os.path.exists(folder):
            os.makedirs(folder)

    return app

3. app/models.py
# 

from datetime import datetime
from app import db
from flask_login import UserMixin
from werkzeug.security import generate_password_hash, check_password_hash

class User(UserMixin, db.Model):
    id = db.Column(db.Integer, primary_key=True)
    username = db.Column(db.String(64), index=True, unique=True)
    email = db.Column(db.String(120), index=True, unique=True)
    password_hash = db.Column(db.String(128))
    created_at = db.Column(db.DateTime, default=datetime.utcnow)
    last_seen = db.Column(db.DateTime, default=datetime.utcnow)
    
    comments = db.relationship('Comment', backref='author', lazy='dynamic')
    bookmarks = db.relationship('Bookmark', backref='user', lazy='dynamic')
    
    def set_password(self, password):
        self.password_hash = generate_password_hash(password)
    
    def check_password(self, password):
        return check_password_hash(self.password_hash, password)
    
    def __repr__(self):
        return f'<User {self.username}>'

class Manga(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    title = db.Column(db.String(140), index=True)
    description = db.Column(db.Text)
    cover_image = db.Column(db.String(140))
    author = db.Column(db.String(140))
    status = db.Column(db.String(20))  # Ongoing, Completed, Hiatus
    created_at = db.Column(db.DateTime, default=datetime.utcnow)
    updated_at = db.Column(db.DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)
    
    chapters = db.relationship('Chapter', backref='manga', lazy='dynamic')
    comments = db.relationship('Comment', backref='manga', lazy='dynamic')
    
    def __repr__(self):
        return f'<Manga {self.title}>'

class Chapter(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    title = db.Column(db.String(140))
    chapter_number = db.Column(db.Float, index=True)
    manga_id = db.Column(db.Integer, db.ForeignKey('manga.id'))
    created_at = db.Column(db.DateTime, default=datetime.utcnow)
    
    pages = db.relationship('Page', backref='chapter', lazy='dynamic')
    comments = db.relationship('Comment', backref='chapter', lazy='dynamic')
    
    def __repr__(self):
        return f'<Chapter {self.chapter_number} of {self.manga.title}>'

class Page(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    page_number = db.Column(db.Integer)
    image_path = db.Column(db.String(140))
    chapter_id = db.Column(db.Integer, db.ForeignKey('chapter.id'))
    
    def __repr__(self):
        return f'<Page {self.page_number} of chapter {self.chapter.chapter_number}>'

class Comment(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    body = db.Column(db.Text)
    created_at = db.Column(db.DateTime, default=datetime.utcnow)
    user_id = db.Column(db.Integer, db.ForeignKey('user.id'))
    manga_id = db.Column(db.Integer, db.ForeignKey('manga.id'))
    chapter_id = db.Column(db.Integer, db.ForeignKey('chapter.id'), nullable=True)
    
    def __repr__(self):
        return f'<Comment by {self.author.username}>'

class Bookmark(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    user_id = db.Column(db.Integer, db.ForeignKey('user.id'))
    manga_id = db.Column(db.Integer, db.ForeignKey('manga.id'))
    chapter_id = db.Column(db.Integer, db.ForeignKey('chapter.id'), nullable=True)
    page_number = db.Column(db.Integer, nullable=True)
    created_at = db.Column(db.DateTime, default=datetime.utcnow)
    notes = db.Column(db.Text, nullable=True)
    
    def __repr__(self):
        return f'<Bookmark for {self.user.username}>'

4. app/auth.py
# 

from flask import Blueprint, render_template, redirect, url_for, flash
from flask_login import login_user, logout_user, login_required
from app import db
from app.models import User
from app.forms import LoginForm, RegistrationForm

bp = Blueprint('auth', __name__)

@bp.route('/login', methods=['GET', 'POST'])
def login():
    form = LoginForm()
    if form.validate_on_submit():
        user = User.query.filter_by(username=form.username.data).first()
        if user is None or not user.check_password(form.password.data):
            flash('Invalid username or password')
            return redirect(url_for('auth.login'))
        login_user(user, remember=form.remember_me.data)
        return redirect(url_for('main.index'))
    return render_template('auth/login.html', title='Sign In', form=form)

@bp.route('/logout')
@login_required
def logout():
    logout_user()
    return redirect(url_for('main.index'))

@bp.route('/register', methods=['GET', 'POST'])
def register():
    form = RegistrationForm()
    if form.validate_on_submit():
        user = User(username=form.username.data, email=form.email.data)
        user.set_password(form.password.data)
        db.session.add(user)
        db.session.commit()
        flash('Congratulations, you are now a registered user!')
        return redirect(url_for('auth.login'))
    return render_template('auth/register.html', title='Register', form=form)

5. app/main.py
# 

from flask import Blueprint, render_template, request, redirect, url_for, flash, current_app, send_from_directory
from flask_login import current_user, login_required
from werkzeug.utils import secure_filename
from app import db
from app.models import Manga, Chapter, Page, Comment, Bookmark
from app.forms import MangaForm, ChapterForm, CommentForm, BookmarkForm
import os
import zipfile
from PIL import Image
import io

bp = Blueprint('main', __name__)

@bp.route('/')
def index():
    recent_manga = Manga.query.order_by(Manga.updated_at.desc()).limit(10).all()
    return render_template('index.html', recent_manga=recent_manga)

@bp.route('/browse')
def browse():
    manga_list = Manga.query.all()
    return render_template('manga/browse.html', manga_list=manga_list)

@bp.route('/manga/<int:manga_id>')
def manga_detail(manga_id):
    manga = Manga.query.get_or_404(manga_id)
    chapters = manga.chapters.order_by(Chapter.chapter_number).all()
    comments = manga.comments.order_by(Comment.created_at.desc()).all()
    form = CommentForm()
    return render_template('manga/detail.html', manga=manga, chapters=chapters, comments=comments, form=form)

@bp.route('/manga/<int:manga_id>/chapter/<int:chapter_id>')
def read_chapter(manga_id, chapter_id):
    manga = Manga.query.get_or_404(manga_id)
    chapter = Chapter.query.get_or_404(chapter_id)
    pages = chapter.pages.order_by(Page.page_number).all()
    
    # Get next and previous chapters
    prev_chapter = Chapter.query.filter(
        Chapter.manga_id == manga_id,
        Chapter.chapter_number < chapter.chapter_number
    ).order_by(Chapter.chapter_number.desc()).first()
    
    next_chapter = Chapter.query.filter(
        Chapter.manga_id == manga_id,
        Chapter.chapter_number > chapter.chapter_number
    ).order_by(Chapter.chapter_number.asc()).first()
    
    return render_template('manga/reader.html', 
                         manga=manga, 
                         chapter=chapter, 
                         pages=pages,
                         prev_chapter=prev_chapter,
                         next_chapter=next_chapter)

@bp.route('/upload', methods=['GET', 'POST'])
@login_required
def upload():
    form = MangaForm()
    if form.validate_on_submit():
        # Handle manga cover upload
        cover_filename = None
        if form.cover_image.data:
            cover_file = form.cover_image.data
            cover_filename = secure_filename(cover_file.filename)
            cover_path = os.path.join(current_app.config['MANGA_COVERS'], cover_filename)
            cover_file.save(cover_path)
            
            # Resize cover image
            img = Image.open(cover_path)
            img.thumbnail((300, 450))
            img.save(cover_path)
        
        manga = Manga(
            title=form.title.data,
            description=form.description.data,
            author=form.author.data,
            status=form.status.data,
            cover_image=cover_filename
        )
        db.session.add(manga)
        db.session.commit()
        
        flash('Manga added successfully!')
        return redirect(url_for('main.manga_detail', manga_id=manga.id))
    
    return render_template('manga/upload.html', form=form)

@bp.route('/manga/<int:manga_id>/add_chapter', methods=['GET', 'POST'])
@login_required
def add_chapter(manga_id):
    manga = Manga.query.get_or_404(manga_id)
    form = ChapterForm()
    
    if form.validate_on_submit():
        chapter = Chapter(
            title=form.title.data,
            chapter_number=form.chapter_number.data,
            manga_id=manga_id
        )
        db.session.add(chapter)
        db.session.commit()
        
        # Handle chapter pages upload
        if form.pages.data:
            pages_file = form.pages.data
            filename = secure_filename(pages_file.filename)
            
            if filename.endswith('.zip') or filename.endswith('.cbz'):
                # Handle ZIP/CBZ file
                zip_path = os.path.join(current_app.config['MANGA_PAGES'], filename)
                pages_file.save(zip_path)
                
                # Extract ZIP file
                extract_folder = os.path.join(current_app.config['MANGA_PAGES'], f'chapter_{chapter.id}')
                os.makedirs(extract_folder, exist_ok=True)
                
                with zipfile.ZipFile(zip_path, 'r') as zip_ref:
                    zip_ref.extractall(extract_folder)
                
                # Process extracted files
                page_number = 1
                for file in sorted(os.listdir(extract_folder)):
                    if file.lower().endswith(('.png', '.jpg', '.jpeg', '.gif')):
                        new_filename = f"{chapter.id}_{page_number}{os.path.splitext(file)[1]}"
                        old_path = os.path.join(extract_folder, file)
                        new_path = os.path.join(current_app.config['MANGA_PAGES'], new_filename)
                        
                        # Rename and resize if needed
                        os.rename(old_path, new_path)
                        
                        # Add page to database
                        page = Page(
                            page_number=page_number,
                            image_path=new_filename,
                            chapter_id=chapter.id
                        )
                        db.session.add(page)
                        page_number += 1
                
                # Clean up
                os.remove(zip_path)
                os.rmdir(extract_folder)
            
            else:
                # Handle single image upload
                page = Page(
                    page_number=1,
                    image_path=filename,
                    chapter_id=chapter.id
                )
                db.session.add(page)
                pages_file.save(os.path.join(current_app.config['MANGA_PAGES'], filename))
        
        db.session.commit()
        flash('Chapter added successfully!')
        return redirect(url_for('main.manga_detail', manga_id=manga_id))
    
    return render_template('manga/add_chapter.html', form=form, manga=manga)

@bp.route('/manga/<int:manga_id>/comment', methods=['POST'])

                user_id=current_user.id,
                manga_id=form.manga_id.data,
                chapter_id=form.chapter_id.data,
                page_number=form.page_number.data,
                notes=form.notes.data
            )
            db.session.add(bookmark)
            db.session.commit()
            flash('Bookmark added!')
    
    return redirect(request.referrer or url_for('main.index'))

6. app/forms.py
# 

from flask_wtf import FlaskForm
from wtforms import StringField, TextAreaField, PasswordField, BooleanField, SubmitField, FloatField, IntegerField, FileField
from wtforms.validators import DataRequired, Email, EqualTo, ValidationError, Length
from app.models import User

class LoginForm(FlaskForm):
    username = StringField('Username', validators=[DataRequired()])
    password = PasswordField('Password', validators=[DataRequired()])
    remember_me = BooleanField('Remember Me')
    submit = SubmitField('Sign In')

class RegistrationForm(FlaskForm):
    username = StringField('Username', validators=[DataRequired()])
    email = StringField('Email', validators=[DataRequired(), Email()])
    password = PasswordField('Password', validators=[DataRequired()])
    password2 = PasswordField('Repeat Password', validators=[DataRequired(), EqualTo('password')])
    submit = SubmitField('Register')

    def validate_username(self, username):
        user = User.query.filter_by(username=username.data).first()
        if user is not None:
            raise ValidationError('Please use a different username.')

    def validate_email(self, email):
        user = User.query.filter_by(email=email.data).first()
        if user is not None:
            raise ValidationError('Please use a different email address.')

class MangaForm(FlaskForm):
    title = StringField('Title', validators=[DataRequired(), Length(max=140)])
    description = TextAreaField('Description')
    author = StringField('Author', validators=[Length(max=140)])
    status = StringField('Status', validators=[Length(max=20)])
    cover_image = FileField('Cover Image')
    submit = SubmitField('Submit')

class ChapterForm(FlaskForm):
    title = StringField('Title', validators=[Length(max=140)])
    chapter_number = FloatField('Chapter Number', validators=[DataRequired()])
    pages = FileField('Pages (ZIP/CBZ or single image)')
    submit = SubmitField('Submit')

class CommentForm(FlaskForm):
    body = TextAreaField('Comment', validators=[DataRequired(), Length(min=1, max=500)])
    submit = SubmitField('Post Comment')

class BookmarkForm(FlaskForm):
    manga_id = IntegerField('Manga ID', validators=[DataRequired()])
    chapter_id = IntegerField('Chapter ID')
    page_number = IntegerField('Page Number')
    notes = TextAreaField('Notes')
    submit = SubmitField('Bookmark')

Setup Instructions
#

  1. Create and activate a virtual environment:

    python -m venv venv
source venv/bin/activate  # On Windows use `venv\Scripts\activate`

  2. Install dependencies:

    pip install -r requirements.txt

  3. Set up the database:

    flask db init
flask db migrate
flask db upgrade

  4. Run the application:

    python run.py

  5. Access the application at http://localhost:5000

The error message:

ERROR [flask_migrate] Error: Target database is not up to date.

means that your Alembic migration scripts and the actual SQLite database schema are out of sync — i.e., the database is not updated to the latest migration.

Here’s how you can resolve this:

✅ Step-by-Step Fix
#

🛠 1. Upgrade the Database Schema
#

Run this command to apply any pending migrations to the database:

flask db upgrade

This will bring your database schema in line with the latest migration scripts.

🧪 2. Check Migration Status
#

You can verify what migrations have been applied:

flask db history

and:

flask db current

This tells you what the current version of the database is and shows the migration history.

💡 Bonus: If flask db upgrade fails
#

If for some reason the upgrade fails (e.g., schema conflicts), you may need to:

  • Inspect your migration script in the migrations/versions/ folder.
  • Manually tweak your database or the migration script.

🤖 Optional Cleanup (only if you’re stuck badly)
#

If your migrations are completely out of sync and you’re in development, you can reset your DB:

rm migrations/ -r
rm app.db  # or your database file
flask db init
flask db migrate -m "initial"
flask db upgrade

⚠️ Warning: This deletes all your existing data and migrations. Only use this in dev/test environments.
