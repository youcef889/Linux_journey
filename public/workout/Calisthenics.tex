\begin{center}
    {\LARGE\bf Beginner Calisthenics Plan (4 Weeks)}\\[1em]
    \textbf{Build Strength, Endurance, and Mobility}
\end{center}

\section{Schedule}
- \textbf{3-4 days per week} (e.g., Monday, Wednesday, Friday)
- Progress each week by increasing reps or difficulty

\section{Workout A: Upper Body Focus}
\textbf{3 Rounds – Rest 30-60 sec between exercises}
\begin{enumerate}
    \item \textbf{Incline Push-ups / Regular Push-ups} – 8-12 reps
    \item \textbf{Negative Push-ups (Lower Slowly)} – 5 reps
    \item \textbf{Knee / Regular Dips (on chair or parallel bars)} – 6-10 reps
    \item \textbf{Superman Hold} – 30 sec
    \item \textbf{Plank} – 30 sec
\end{enumerate}

\section{Workout B: Lower Body \\ Core Focus}
\textbf{3 Rounds – Rest 30-60 sec between exercises}
\begin{enumerate}
    \item \textbf{Bodyweight Squats} – 12-15 reps
    \item \textbf{Lunges (each leg)} – 8-12 reps
    \item \textbf{Glute Bridges} – 10-15 reps
    \item \textbf{Leg Raises / Knee Tucks} – 10 reps
    \item \textbf{Side Plank (each side)} – 20 sec
\end{enumerate}

\section{Optional: Mobility \\ Active Recovery (On Rest Days)}
- Stretching (hips, shoulders, hamstrings)
- Walking or light cardio

\section{Progression Plan (Over 4 Weeks)}
- \textbf{Week 1-2:} Start slow, focus on form
- \textbf{Week 3:} Increase reps (by 2-3 per exercise)
- \textbf{Week 4:} Try harder variations (e.g., incline → regular push-ups)

\section{Tracking Progress}
- Note reps completed each workout
- Aim to improve reps or move to harder variations
