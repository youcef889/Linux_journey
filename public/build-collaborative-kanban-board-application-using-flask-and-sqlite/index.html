Building a Collaborative Kanban Board with Flask and SQLite
#

I’ll help you create a simple but functional Kanban board application that can run on your Alpine Linux server. This will include multiple columns (To Do, In Progress, Done) and allow basic task management.

Project Structure
# 

/kanban-app
  ├── app.py              # Main application file
  ├── requirements.txt    # Python dependencies
  ├── static/             # CSS/JS files
  ├── templates/          # HTML templates
  ├── instance/           # Database files
  └── .env                # Environment variables

# 

# Create project directory

source venv/bin/activate

Step 2: Database Models (app.py)
# 

from flask import Flask, render_template, request, jsonify, redirect, url_for
from flask_sqlalchemy import SQLAlchemy
from datetime import datetime
import os

app = Flask(__name__)
app.config['SECRET_KEY'] = 'your-secret-key'
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///' + os.path.join(app.instance_path, 'kanban.db')
app.config['SQLALCHEMY_TRACK_MODIFICATIONS'] = False
db = SQLAlchemy(app)

# Create the instance folder if it doesn't exist
os.makedirs(app.instance_path, exist_ok=True)

class Task(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    title = db.Column(db.String(100), nullable=False)
    description = db.Column(db.Text)
    status = db.Column(db.String(20), default='todo')  # todo, in_progress, done
    created_at = db.Column(db.DateTime, default=datetime.utcnow)
    position = db.Column(db.Integer, default=0)

    def to_dict(self):
        return {
            'id': self.id,
            'title': self.title,
            'description': self.description,
            'status': self.status,
            'position': self.position
        }

@app.before_first_request
def create_tables():
    db.create_all()

@app.route('/')
def index():
    tasks = {
        'todo': Task.query.filter_by(status='todo').order_by(Task.position).all(),
        'in_progress': Task.query.filter_by(status='in_progress').order_by(Task.position).all(),
        'done': Task.query.filter_by(status='done').order_by(Task.position).all()
    }
    return render_template('index.html', tasks=tasks)

@app.route('/api/tasks', methods=['POST'])
def create_task():
    data = request.get_json()
    task = Task(
        title=data['title'],
        description=data.get('description', ''),
        status=data.get('status', 'todo')
    )
    db.session.add(task)
    db.session.commit()
    return jsonify(task.to_dict()), 201

@app.route('/api/tasks/<int:task_id>', methods=['PUT'])
def update_task(task_id):
    task = Task.query.get_or_404(task_id)
    data = request.get_json()
    
    if 'title' in data:
        task.title = data['title']
    if 'description' in data:
        task.description = data['description']
    if 'status' in data:
        task.status = data['status']
    if 'position' in data:
        task.position = data['position']
    
    db.session.commit()
    return jsonify(task.to_dict())

@app.route('/api/tasks/<int:task_id>', methods=['DELETE'])
def delete_task(task_id):
    task = Task.query.get_or_404(task_id)
    db.session.delete(task)
    db.session.commit()
    return jsonify({'message': 'Task deleted'}), 200

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000, debug=True)

        // Initialize Sortable for drag and drop
        document.querySelectorAll('.task-list').forEach(list => {
            new Sortable(list, {
                group: 'kanban',
                animation: 150,
                onEnd: function(evt) {
                    const taskId = evt.item.dataset.taskId;
                    const newStatus = evt.to.dataset.status;
                    const newPosition = Array.from(evt.to.children).indexOf(evt.item);
                    
                    fetch(`/api/tasks/${taskId}`, {
                        method: 'PUT',
                        headers: {
                            'Content-Type': 'application/json',
                        },
                        body: JSON.stringify({
                            status: newStatus,

            });

        // Add Task Modal

                document.getElementById('taskStatus').value = status;
                const modal = new bootstrap.Modal(document.getElementById('taskModal'));
                modal.show();

        // Save Task
        document.getElementById('saveTaskBtn').addEventListener('click', () => {

            const description = document.getElementById('taskDescription').value;
            const status = document.getElementById('taskStatus').value;
            
            fetch('/api/tasks', {

                headers: {
                    'Content-Type': 'application/json',

                    title: title,
                    description: description,
                    status: status

Features Implemented
#

  1. Three-column Kanban board (To Do, In Progress, Done)
  2. Drag and drop functionality between columns
  3. Add new tasks with title and description
  4. Move tasks between columns with buttons
  5. Delete tasks from the board
  6. Persistent storage using SQLite
  7. Responsive design that works on different screen sizes

Performance Considerations for Your Hardware
#

  1. Limit concurrent users: Your Pentium E5700 with 2GB RAM should handle 2-3 concurrent users comfortably
  2. Disable debug mode in production: 
    app.run(host='0.0.0.0', port=5000, debug=False)
  3. Use Gunicorn for production (as shown in previous instructions)
  4. Keep tasks minimal - this is a lightweight implementation
